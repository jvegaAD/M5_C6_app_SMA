CREATE TABLE Reporte (
    id SERIAL PRIMARY KEY,
    PeriodoInforme DATE NOT NULL,
    RUT VARCHAR(20) NOT NULL,
    RazonSocial VARCHAR(255) NOT NULL,
    Planta VARCHAR(255) NOT NULL,
    PuntoDeDescarga VARCHAR(255) NOT NULL,
    CuerpoReceptor VARCHAR(255) NOT NULL,
    Norma VARCHAR(50) NOT NULL,
    Muestra INTEGER NOT NULL,
    Parametro VARCHAR(255) NOT NULL,
    Unidad VARCHAR(50) NOT NULL,
    ValorReportado NUMERIC NOT NULL,
    TipoDeControl VARCHAR(100) NOT NULL,
    Laboratorio VARCHAR(255) NOT NULL,
    UnidadFiscalizable VARCHAR(255) NOT NULL,
    Departamento VARCHAR(255) NOT NULL,
    ComunaId INTEGER NOT NULL,
    Comuna VARCHAR(255) NOT NULL,
    NombreCategoria VARCHAR(255) NOT NULL,
    NombreSubCategoria VARCHAR(255) NOT NULL,
    Direccion VARCHAR(255) NOT NULL,
    NumeroRCA INTEGER NOT NULL,
    FechaRCA DATE NOT NULL
);
